별 어려움 없었음.
pop 할 때 실수해서 인덱스 초과가 났었음.
파이썬 기초가 더 필요함.

from collections import deque

def solution(priorities, location):
    q_wait = deque()
    success = False
    order = 1
    priors = []
    max_prior = -1
    answer = 0

    # 대기열 채워넣기
    for idx, doc in enumerate(priorities):
        q_wait.append((idx, doc))
        priors.append(doc)

    priors.append(-1)
    priors.sort(reverse=True)
    max_prior = priors.pop(0)

    # 목표 문서를 프린트할 때까지 액션 반복
    while not success:
        doc = q_wait.popleft()

        # 뽑은 문서가 최고 우선순위일 때
        if doc[1] >= max_prior:
            max_prior = priors.pop(0)
            # 그것이 목표 문서일 때 성공.
            if doc[0] == location:
                success = True
                continue
            # 아니면 인쇄 횟수만 카운트
            else:
                order += 1
        # 최고 우선순위가 아닐 때
        else:
            q_wait.append(doc)
    answer = order
    return answer
