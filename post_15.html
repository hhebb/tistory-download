총평

어려움 없이 해결.

def solution(prices):
    answer = []

    for i in range(len(prices) - 1):
        current_price = prices[i]
        duration = 0

        for j in range(i + 1, len(prices)):
            # 체크하기 전 시간 더해준다.
            duration += 1
            # 가격이 떨어지지 않으면 다음루프 진행
            if current_price <= prices[j]:
                continue
            else:
                # 가격이 떨어졌으면 종료
                break
        answer.append(duration)

    # 마지막 시간은 무조건 0
    answer.append(0)
    return answer
