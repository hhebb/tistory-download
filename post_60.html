총평

로직의 단계가 나뉘어져 있지만 하나하나가 난이도가 높지 않아서 전체적으로 평이했다.

def solution(n, t, m, p):
    answer = ''
    tubes = []
    order = 0
    num = 0
    match = {10:'A', 11:'B', 12:'C', 13:'D', 14:'E', 15:'F'}

    while len(tubes) < t:

        num_copy = num
        tmp_res = ''
        while num_copy >= n:
            add_num = num_copy % n
            if add_num >= 10:
                add_num = match[add_num]
            else:
                add_num = str(add_num)
            tmp_res = add_num + tmp_res
            num_copy = num_copy // n
            #print(num_copy)
        if num_copy != 0:
            add_num = num_copy % n
            if add_num >= 10:
                add_num = match[add_num]
            else:
                add_num = str(add_num)
            tmp_res = add_num + tmp_res

        if num == 0:
            tmp_res = '0'

        for tmp in tmp_res:
            #print(order, tmp)
            if order % m == p - 1:
                tubes.append(tmp)
            order += 1

        num += 1

    tubes = tubes[:t]
    answer = ''.join(tubes)
    return answer
