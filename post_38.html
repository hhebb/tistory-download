큰 어려움 없이 해결할 수 있는 문제.

from collections import deque

def solution(s):
    answer = True
    stack = deque()

    for idx, bracket in enumerate(s):
        if bracket == '(':
            if idx == len(s) - 1:
                answer = False
                break
            else:
                stack.append(bracket)
        else:
            if len(stack) == 0:
                answer = False
                break
            else:
                stack.popleft()

    if len(stack) != 0:
        answer = False

    return answer
