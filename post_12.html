좌표로 접근하니 간편해짐.
거리 구하는 함수, 숫자-좌표 매핑 함수를 미리 만든것은 좋았음.

def mapping(num):
    pos = []
    if num == 1:pos = [3, 0]
    elif num == 2:pos = [3, 1]
    elif num == 3:pos = [3, 2]
    elif num == 4:pos = [2, 0]
    elif num == 5:pos = [2, 1]
    elif num == 6:pos = [2, 2]
    elif num == 7:pos = [1, 0]
    elif num == 8:pos = [1, 1]
    elif num == 9:pos = [1, 2]
    elif num == 0:pos = [0, 1]
    return pos

def distance(pos_1, pos_2):
    dx = abs(pos_1[0] - pos_2[0])
    dy = abs(pos_1[1] - pos_2[1])
    return dx + dy

def solution(numbers, hand):
    answer = ''
    left = [0, 0]
    right = [0, 2]

    for num in numbers:
        if num in numbers:
            # 1 4 7 이면 왼손으로 누르고 이동.
            if num in [1, 4, 7]:
                answer += 'L'
                left = mapping(num)
            # 3 6 9 이면 오른손으로 누르고 이동.
            elif num in [3, 6, 9]:
                answer += 'R'
                right = mapping(num)
            # 2 4 8 0 이면 거리 판단
            else:
                # 오른쪽이 멀면 왼손으로 누르고 이동
                if(distance(right, mapping(num)) > distance(left, mapping(num))):
                    answer += 'L'
                    left = mapping(num)
                # 왼쪽이 멀면 오른손으로 누르고 이동
                elif(distance(right, mapping(num)) < distance(left, mapping(num))):
                    answer += 'R'
                    right = mapping(num)
                # 거리가 같으면 오른손잡이 여부 판단하고 누르고 이동
                else:
                    if(hand == 'right'):
                        answer += 'R'
                        right = mapping(num)
                    else:
                        answer += 'L'
                        left = mapping(num)
    return answer
