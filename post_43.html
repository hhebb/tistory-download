def solution(n):
    answer = 0

    for i in range(1, int((n + 1)/2) + 1):
        total = 0
        for j in range(i, int((n + 1)/2) + 1):
            total += j
            if total == n:
                answer += 1
            elif total > n:
                break
    answer += 1
    return answer
