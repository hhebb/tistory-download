def solution(skill, skill_trees):
    answer = 0
    info = []

    # 스킬 정보 
    for idx, s in enumerate(skill):
        info.append(s)

    for tree in skill_trees:
        order = 0
        possible = True

        for s in tree:
            # 스킬정보에 있는 스킬일 때
            if s in skill:
                # 순서에 맞을 때 순서 올려주고 다음 루프 진행
                if s == info[order]:
                    order += 1
                    pass
                # 아니면 불가능
                else:
                    possible = False
                    break
        if possible:
            answer += 1
    return answer
