예외처리는 어렵지 않았음.

 

cache = [[0] * i for i in range(1, 501)]

def solution(triangle):
    answer = 0
    cache[0][0] = triangle[0][0]
    
    for row in range(1, len(triangle)):
        for col in range(row+1):
            if col == 0:
                cache[row][col] = cache[row-1][col] + triangle[row][col]
            elif col == row:
                cache[row][col] = cache[row-1][col-1] + triangle[row][col]
            else:
                cache[row][col] = max(cache[row-1][col-1], cache[row-1][col]) + triangle[row][col]
        
    answer = max(cache[len(triangle) - 1])
    return answer
