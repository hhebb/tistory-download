check = [False] * 201
graph = [[] for i in range(201)]

def dfs(node):
    if check[node]:
        return
    else:
        check[node] = True
        
    for next_node in graph[node]:
        dfs(next_node)

def solution(n, computers):
    answer = 0    
    
    for idx, info in enumerate(computers):
        for idx2, node in enumerate(info):
            if node == 1 and idx != idx2:
                graph[idx].append(idx2)
    
    for node in range(n):
        if check[node]:
            pass
        else:
            answer += 1
            dfs(node)
        
    return answer
