문자열 합치기 연산

총평

문제가 2 단계라 겁먹을 수도 있음.
빨리 지면에 풀어봐야함.

def decode(n, num):
    original = [0] * n

    i = n - 1
    while num >= 1:
        original[i] = num % 2
        num = num // 2
        i -= 1
    return original

def solution(n, arr1, arr2):
    answer = []

    # 지도 초기화
    map_1 = [[0] *n for _ in range(n)]
    map_2 = [[0] *n for _ in range(n)]
    map_target = [[0] *n for _ in range(n)]

    # 지도 1 해석
    i = 0
    for num in arr1:
        map_1[i] = decode(n, num)
        i += 1

    i = 0
    # 지도 2 해석
    for num in arr2:
        map_2[i] = decode(n, num)
        i += 1

    # 지도 합성 및 문자 변환
    for r in range(n):
        word = ''
        for c in range(n):
            #map_target[r][c] = map_1[r][c] or map_2[r][c]
            result = map_1[r][c] or map_2[r][c]
            if result == 1:
                word += '#'
            else:
                word += ' '
        answer.append(word)
    # 지도 문자 변환

    return answer
